## Project Detection Logic

```javascript
// Pseudo-code for project type detection
function detectProjectType(files) {
  if (files.includes("package.json")) {
    if (files.includes("next.config.js")) return "Next.js";
    if (files.includes("vite.config.js")) return "Vite";
    if (files.includes("angular.json")) return "Angular";
    return "Node.js";
  }
  if (files.includes("requirements.txt")) return "Python";
  if (files.includes("pom.xml")) return "Java";
  if (files.includes("go.mod")) return "Go";
  return "Generic";
}
```
